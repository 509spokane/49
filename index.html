[\s\S]*?<\/div>/gi; const boxes = html.match(boxRegex); if (boxes) { boxes.forEach(box => { const labelMatch = box.match(/

(.*?)<\/h4>/i); const valueMatch = box.match(/

(.*?)<\/h3>/i); if (labelMatch && valueMatch) { const label = labelMatch[1].trim(); const value = valueMatch[1].trim().replace(/"/g, ""); switch(label) { case "12 Hours": snowData["12h"] = value; break; case "24 Hours": snowData["24h"] = value; break; case "48 Hours": snowData["48h"] = value; break; case "72 Hours": snowData["72h"] = value; break; case "Snow Depth": snowData["summitDepth"] = value; break; } } }); } // --- Last Updated (rounded, with 11pm exception) --- const updatedMatch = html.match(/Last Updated:\s*([0-9\/]+)\s*at\s*([0-9:]+)(am|pm)/i); if (updatedMatch) { const date = updatedMatch[1]; // e.g., 03/22/26 let hour = parseInt(updatedMatch[2].split(":")[0]); let minutes = parseInt(updatedMatch[2].split(":")[1]); let period = updatedMatch[3].toLowerCase(); // Convert to 24-hour if (period === "pm" && hour !== 12) hour += 12; if (period === "am" && hour === 12) hour = 0; // Round minutes if (minutes >= 30) hour += 1; // Exception: if hour == 24, force 23 (11pm) if (hour >= 24) hour = 23; // Convert back to 12-hour let displayHour = hour % 12; if (displayHour === 0) displayHour = 12; let displayPeriod = hour >= 12 ? "pm" : "am"; // Remove year const shortDate = date.split('/').slice(0,2).join('/'); snowData["updated"] = `${shortDate} ${displayHour}${displayPeriod}`; } return new Response(JSON.stringify(snowData), { headers: { "Content-Type": "application/json", "Access-Control-Allow-Origin": "*" } }); } catch (err) { return new Response(JSON.stringify({error: "Data Error"}), { headers: { "Content-Type": "application/json", "Access-Control-Allow-Origin": "*" } }); } }
